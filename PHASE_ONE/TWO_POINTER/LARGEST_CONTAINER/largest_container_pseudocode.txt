# Input: List[int]
# The integer values in the array represent the height of the array
# Function will be determining the max amount of water that can be stored in a
container without overflow

FUNCTION largest_container(List[int]) -> int:
    # INITIALIZE POINTERS AT BOTH ENDS OF ARRAY
    left, right = first, last

    FIND MAX VOLUME OF WATER STORED IN WIDEST CONTAINER
    # STORE THE MAX_VOLUME
    max_water = MAX_VOLUME
    MOVE LEFT FORWARD IF LEFT < RIGHT OR MOVE RIGHT FORWARD IF RIGHT < LEFT
    # CONDITIONAL LOOP TO PREVENT POINTERS FROM COLLIDING
    WHILE left < right:
        GET HEIGHT (min(List[left], List[right]))
        GET WIDTH (right(index position) - left(index position))
        possible_new_max_water = HEIGHT * WIDTH
        IF HEIGHT * WIDTH > max_water:
            max_water = possible_new_max_water
        # CHECK WHICH POINTER TO MOVE
        IF List[left] < List[right]:
            MOVE LEFT POINTER RIGHT
        IF List[right] < List[left]:
            MOVE RIGHT POINTER LEFT

    RETURN max_water
